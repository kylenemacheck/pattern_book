public interface cookie
{
  public void bake();
}
public class ChoclateChip implements Cookie
{
  public void bake()
  {
    System.out.println("Choclate chip cookie is ready.");
  }
}
public class Snickerdoodle implements Cookie
{
  public void bake()
  {
    System.out.println("Snickerdoodle cookie is ready.");
  }
}
public class DoubleChoclate implements Cookie
{
  public void bake()
  {
    System.out.println("Double choclate cookie is ready.");
  }
}

Facade Class
public class Oven
{
  private Cookie choclateChip;
  private Cookie snickerdoodle;
  private Cookie doubleChoclate;
  public Oven()
  {
    choclateChip = new ChoclateChip();
    snickerDoodle = new Snickerdoodle();
    doubleChoclate = new DoubleChoclate();
  }
  public void bakeChoclateChip()
  {
    choclateChip.bake();
  }
  public void bakeSnickerdoodle()
  {
    snickerdoodle.bake();
  }
  public void bakeDoubleChoclate()
  {
    doubleChoclate.bake();
  }
}
