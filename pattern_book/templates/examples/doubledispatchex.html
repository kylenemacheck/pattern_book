Before Pattern Use.
public class Cup
{
  public void make(Handle handle, Stem stem)
  {
    if (stem != null)
      System.out.println("Makes a Coffe Cup");
    else if (handle != null)
      System.out.println("Makes a Wine Glass");
    else
      System.out.println("Not a Cup");
  }
}

After Pattern Use.
public class CoffeeCup extents Cup public class WineGlass extends Cup
{ {
  public void make(Handle handle)   public void make(Stem stem)
  {   {
    System.out.println("Makes a Coffee Cup");     System.out.println("Makes a Wine Glass");
  }   }
} }
